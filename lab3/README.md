class Product {
  // Properties
  String _id; // private (_) ‡∏ï‡∏≤‡∏°‡πÇ‡∏à‡∏ó‡∏¢‡πå [cite: 12]
  String name;
  double _price; // ‡πÉ‡∏ä‡πâ _price ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÄ‡∏Å‡πá‡∏ö‡∏Ñ‡πà‡∏≤ ‡πÅ‡∏•‡∏∞‡πÉ‡∏ä‡πâ getter/setter ‡∏Ñ‡∏∏‡∏°
  int? stock; // nullable (?) ‡∏ï‡∏≤‡∏°‡πÇ‡∏à‡∏ó‡∏¢‡πå [cite: 15]

  // Constructor
  // ‡πÉ‡∏ä‡πâ named parameters ‡πÅ‡∏•‡∏∞ required ‡∏ï‡∏≤‡∏°‡πÇ‡∏à‡∏ó‡∏¢‡πå [cite: 17]
  Product({
    required String id,
    required this.name,
    required double price,
    this.stock,
  }) : _id = id,
       _price = price;

  // Getter: ‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡πà‡∏≤ _id [cite: 19]
  String get id => _id;

  // Getter: ‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡πà‡∏≤ _price
  double get price => _price;

  // Setter: price [cite: 20]
  // ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏• ‡∏ï‡πâ‡∏≠‡∏á‡∏°‡∏≤‡∏Å‡∏Å‡∏ß‡πà‡∏≤ 0 ‡πÄ‡∏ó‡πà‡∏≤‡∏ô‡∏±‡πâ‡∏ô
  set price(double newPrice) {
    if (newPrice > 0) {
      _price = newPrice;
    } else {
      print('‡∏£‡∏≤‡∏Ñ‡∏≤‡πÑ‡∏°‡πà‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á (‡∏ï‡πâ‡∏≠‡∏á > 0) -> ‡πÑ‡∏°‡πà‡πÄ‡∏õ‡∏•‡∏µ‡πà‡∏¢‡∏ô‡∏Ñ‡πà‡∏≤'); // 
    }
  }

  // Method: applyDiscount [cite: 23]
  void applyDiscount(double percent) {
    if (percent >= 0 && percent <= 100) {
      _price = _price - (_price * (percent / 100));
    } else {
      print('‡πÄ‡∏õ‡∏≠‡∏£‡πå‡πÄ‡∏ã‡πá‡∏ô‡∏ï‡πå‡∏™‡πà‡∏ß‡∏ô‡∏•‡∏î‡πÑ‡∏°‡πà‡∏ñ‡∏π‡∏Å‡∏ï‡πâ‡∏≠‡∏á');
    }
  }

  // Method: restock [cite: 26]
  void restock(int amount) {
    // ‡∏ñ‡πâ‡∏≤ stock ‡πÄ‡∏õ‡πá‡∏ô null ‡πÉ‡∏´‡πâ‡∏ñ‡∏∑‡∏≠‡∏ß‡πà‡∏≤‡πÄ‡∏£‡∏¥‡πà‡∏°‡∏à‡∏≤‡∏Å 0 ‡πÅ‡∏•‡πâ‡∏ß‡∏ö‡∏ß‡∏Å‡πÄ‡∏û‡∏¥‡πà‡∏°
    stock = (stock ?? 0) + amount; // 
  }

  // Method: showInfo [cite: 29]
  void showInfo() {
    print('ID: $_id');
    print('Name: $name');
    print('Price: $_price');
    // ‡∏ñ‡πâ‡∏≤ stock ‡πÄ‡∏õ‡πá‡∏ô null ‡πÉ‡∏´‡πâ‡πÅ‡∏™‡∏î‡∏á "‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏£‡∏∞‡∏ö‡∏∏‡∏™‡∏ï‡πá‡∏≠‡∏Å" [cite: 31]
    print('Stock: ${stock ?? "‡∏¢‡∏±‡∏á‡πÑ‡∏°‡πà‡∏£‡∏∞‡∏ö‡∏∏‡∏™‡∏ï‡πá‡∏≠‡∏Å"}');
  }
